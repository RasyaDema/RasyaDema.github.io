import pygame import random # Inisialisasi pygame pygame.init() # Ukuran layar WIDTH, HEIGHT = 400, 600 screen = pygame.display.set_mode((WIDTH, HEIGHT)) pygame.display.set_caption("Avoid the Falling Objects") # Warna WHITE = (255, 255, 255) BLACK = (0, 0, 0) RED = (255, 0, 0) # Pengaturan pemain player_size = 50 player_x = WIDTH // 2 - player_size // 2 player_y = HEIGHT - player_size - 10 player_speed = 5 # Pengaturan musuh enemy_size = 50 enemy_speed = 5 enemy_list = [] # Fungsi untuk membuat musuh baru def create_enemy(): x_pos = random.randint(0, WIDTH - enemy_size) y_pos = 0 - enemy_size enemy_list.append([x_pos, y_pos]) # Fungsi untuk menampilkan musuh di layar def draw_enemies(enemy_list): for enemy in enemy_list: pygame.draw.rect(screen, RED, (enemy[0], enemy[1], enemy_size, enemy_size)) # Fungsi untuk menggerakkan musuh def update_enemy_positions(enemy_list): global score for enemy in enemy_list: enemy[1] += enemy_speed if enemy[1] >= HEIGHT: enemy_list.remove(enemy) score += 1 # Fungsi untuk mendeteksi tabrakan def detect_collision(player_pos, enemy_pos): p_x, p_y = player_pos e_x, e_y = enemy_pos if (e_x < p_x < e_x + enemy_size or e_x < p_x + player_size < e_x + enemy_size) and \ (e_y < p_y < e_y + enemy_size or e_y < p_y + player_size < e_y + enemy_size): return True return False # Game loop clock = pygame.time.Clock() score = 0 running = True while running: screen.fill(WHITE) # Event loop for event in pygame.event.get(): if event.type == pygame.QUIT: running = False # Menggerakkan pemain keys = pygame.key.get_pressed() if keys[pygame.K_LEFT] and player_x - player_speed > 0: player_x -= player_speed if keys[pygame.K_RIGHT] and player_x + player_size + player_speed < WIDTH: player_x += player_speed # Membuat musuh baru if random.randint(1, 20) == 1: create_enemy() # Mengupdate posisi musuh update_enemy_positions(enemy_list) # Cek tabrakan for enemy in enemy_list: if detect_collision((player_x, player_y), enemy): running = False # Gambar pemain pygame.draw.rect(screen, BLACK, (player_x, player_y, player_size, player_size)) # Gambar musuh draw_enemies(enemy_list) # Tampilkan skor font = pygame.font.SysFont("monospace", 35) score_text = font.render("Score: " + str(score), True, BLACK) screen.blit(score_text, (10, 10)) pygame.display.flip() clock.tick(30) pygame.quit()
